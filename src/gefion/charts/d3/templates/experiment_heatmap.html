g.append("g").attr("class", "axis") .call(d3.axisLeft(y)); // Axis labels g.append("text").attr("x", w / 2).attr("y", h + 45) .attr("text-anchor", "middle").attr("fill", "var(--color-text)") .attr("font-size", "12px").text(chartConfig.x_label || "X"); // Cells g.selectAll(".cell").data(chartData).join("rect") .attr("class", "cell") .attr("x", d => x(String(d.x))) .attr("y", d => y(String(d.y))) .attr("width", x.bandwidth()) .attr("height", y.bandwidth()) .attr("fill", d => color(d.value)) .attr("stroke", "#e0e0e0") .on("mousemove", function(event, d) { tip.show( `${chartConfig.x_label || "X"}: ${d.x}
